Definition ProCGroups.GroupTheory.IsNormalIn Source 
def IsNormalIn (H K : Subgroup G) : Prop :=
  H ≤ K ∧ ∀ ⦃k h : G⦄, k ∈ K → h ∈ H → k * h * k⁻¹ ∈ H
