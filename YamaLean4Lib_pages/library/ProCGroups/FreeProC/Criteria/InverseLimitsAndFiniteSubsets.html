def IsSmallestClosedNormalSubgroupContaining
    {G : Type u} [Group G] [TopologicalSpace G]
    (A : Set G) (N : Subgroup G) : Prop :=
  N.Normal ∧
    IsClosed ((N : Set G)) ∧

    ∀ M : Subgroup G, M.Normal → IsClosed ((M : Set G)) → A ⊆ (M : Set G) → N ≤ M
