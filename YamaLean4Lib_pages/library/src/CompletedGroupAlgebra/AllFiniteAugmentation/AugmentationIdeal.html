35def completedGroupAlgebraCanonicalAugmentationIdeal (R : Type u) (G : Type v) [CommRing R]
36 [TopologicalSpace R] [IsTopologicalRing R] [Group G] [TopologicalSpace G]
37 [IsTopologicalGroup G] : Ideal (Carrier R G) :=
