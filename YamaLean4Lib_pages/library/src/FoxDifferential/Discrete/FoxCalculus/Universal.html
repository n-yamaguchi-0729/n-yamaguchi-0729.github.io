92/-- The universal relative differential on a free group satisfies the inverse rule. -/
93theorem relativeFreeGroupDifferential_inv (w : FreeGroup X) :
94 universalDifferential ψ w⁻¹ =
95 -((MonoidAlgebra.of ℤ H (ψ w⁻¹) : GroupRing H) • universalDifferential ψ w) := by
96 have h := universalDifferential_mul_inv_right ψ w⁻¹
97 rw [eq_neg_iff_add_eq_zero]
98 simpa using h

100/-- The relative Fox-coordinate formula recovers the universal differential in `A_ψ`. -/
101theorem relativeFreeFoxCoordinatesLinearMap_derivative (w : FreeGroup X) :
102 relativeFreeFoxCoordinatesLinearMap (H := H) X ψ
103 (relativeFreeGroupFoxDerivative (H := H) X ψ w) =
104 universalDifferential ψ w := by
105 induction w using FreeGroup.induction_on with
106 | C1 =>
107 simp only [relationSubmodule_eq_crossedDifferentialRelationSubmodule, relativeFreeGroupFoxDerivative_one,
108 map_zero, universalDifferential_one]
109 | of x =>
110 simp only [relationSubmodule_eq_crossedDifferentialRelationSubmodule, relativeFreeGroupFoxDerivative_of,
111 relativeFreeFoxCoordinatesLinearMap_single]
112 | inv_of x hx =>
113 rw [relativeFreeGroupFoxDerivative_inv, map_neg, map_smul, hx]
114 exact (relativeFreeGroupDifferential_inv (H := H) X ψ (FreeGroup.of x)).symm
115 | mul x y hx hy =>
116 rw [relativeFreeGroupFoxDerivative_mul, map_add, map_smul, hx, hy]
117 simpa using (universalDifferential_mul ψ x y).symm
