83/-- Finite-stage Fox fundamental formula for a free-group word. -/
84theorem finiteFoxStageDerivative_fundamental_formula
85 [Fintype X] (w : FreeGroup X) :
86 MonoidAlgebra.of (ModNCompletedCoeff n)
87 (finiteFoxStageTargetQuotient (X := X) N) (QuotientGroup.mk' N w) - 1 =
88 ∑ i : X,
89 finiteFoxStageDerivative (X := X) N n i w *
90 (MonoidAlgebra.of (ModNCompletedCoeff n)
91 (finiteFoxStageTargetQuotient (X := X) N)
92 (QuotientGroup.mk' N (FreeGroup.of i)) - 1) := by
93 induction w using FreeGroup.induction_on with
94 | C1 =>
95 simp only [QuotientGroup.mk'_apply, MonoidAlgebra.of_apply, MonoidAlgebra.one_def,
96 sub_self, finiteFoxStageDerivative, finiteFoxStageDerivativeVector, map_one, FiniteFoxStageSemidirect.one_left,
97 Pi.zero_apply, zero_mul, Finset.sum_const_zero]
98 | of x =>
99 simp only [QuotientGroup.mk'_apply, MonoidAlgebra.of_apply, MonoidAlgebra.one_def, finiteFoxStageDerivative,
100 finiteFoxStageDerivativeVector_of, Pi.single_apply, ite_mul, zero_mul, Finset.sum_ite_eq', Finset.mem_univ,
101 ↓reduceIte]
102 have hone :
103 (MonoidAlgebra.single
104 (1 : finiteFoxStageTargetQuotient (X := X) N)
105 (1 : ModNCompletedCoeff n) :
106 finiteFoxStageTargetGroupAlgebra (X := X) N n) = 1 := by
107 simp only [MonoidAlgebra.one_def]
108 rw [hone, one_mul]
109 | inv_of x hx =>
110 simp only [QuotientGroup.mk'_apply, MonoidAlgebra.of_apply, MonoidAlgebra.one_def,
111 finiteFoxStageDerivative, finiteFoxStageDerivativeVector, finiteFoxStageLift, map_inv, FreeGroup.lift_apply_of,
112 FiniteFoxStageSemidirect.inv_left, Pi.neg_apply, Pi.smul_apply, Pi.single_apply, smul_eq_mul, mul_ite,
113 MonoidAlgebra.single_mul_single, mul_one, mul_zero, neg_mul, ite_mul, zero_mul, Finset.sum_neg_distrib,
114 Finset.sum_ite_eq', Finset.mem_univ, ↓reduceIte]
115 change
116 MonoidAlgebra.single ((QuotientGroup.mk' N (FreeGroup.of x))⁻¹)
117 (1 : ModNCompletedCoeff n) -
118 MonoidAlgebra.single (1 : finiteFoxStageTargetQuotient (X := X) N)
119 (1 : ModNCompletedCoeff n) =
120 - (MonoidAlgebra.single ((QuotientGroup.mk' N (FreeGroup.of x))⁻¹)
121 (1 : ModNCompletedCoeff n) *
122 (MonoidAlgebra.single (QuotientGroup.mk' N (FreeGroup.of x))
123 (1 : ModNCompletedCoeff n) -
124 MonoidAlgebra.single (1 : finiteFoxStageTargetQuotient (X := X) N)
125 (1 : ModNCompletedCoeff n)))
126 have hone :
127 (MonoidAlgebra.single
128 (1 : finiteFoxStageTargetQuotient (X := X) N)
129 (1 : ModNCompletedCoeff n) :
130 finiteFoxStageTargetGroupAlgebra (X := X) N n) = 1 := by
131 simp only [MonoidAlgebra.one_def]
132 have hmul :
133 (MonoidAlgebra.single ((QuotientGroup.mk' N (FreeGroup.of x))⁻¹)
134 (1 : ModNCompletedCoeff n) *
135 MonoidAlgebra.single (QuotientGroup.mk' N (FreeGroup.of x))
136 (1 : ModNCompletedCoeff n) :
137 finiteFoxStageTargetGroupAlgebra (X := X) N n) = 1 := by
138 simp only [QuotientGroup.mk'_apply, MonoidAlgebra.single_mul_single, inv_mul_cancel, mul_one,
139 MonoidAlgebra.one_def]
140 rw [hone, mul_sub, hmul, mul_one]
141 noncomm_ring
142 | mul u v hu hv =>
143 let gu : finiteFoxStageTargetGroupAlgebra (X := X) N n :=
144 MonoidAlgebra.of (ModNCompletedCoeff n)
145 (finiteFoxStageTargetQuotient (X := X) N) (QuotientGroup.mk' N u)
146 let gv : finiteFoxStageTargetGroupAlgebra (X := X) N n :=
147 MonoidAlgebra.of (ModNCompletedCoeff n)
148 (finiteFoxStageTargetQuotient (X := X) N) (QuotientGroup.mk' N v)
149 have hmul :
150 MonoidAlgebra.of (ModNCompletedCoeff n)
151 (finiteFoxStageTargetQuotient (X := X) N) (QuotientGroup.mk' N (u * v)) =
152 gu * gv := by
153 simp only [QuotientGroup.mk'_apply, QuotientGroup.mk_mul, MonoidAlgebra.of_apply,
154 MonoidAlgebra.single_mul_single, mul_one, gu, gv]
155 calc
156 MonoidAlgebra.of (ModNCompletedCoeff n)
157 (finiteFoxStageTargetQuotient (X := X) N) (QuotientGroup.mk' N (u * v)) - 1
158 = (gu - 1) + gu * (gv - 1) := by
159 rw [hmul]
160 noncomm_ring
161 _ =
162 (∑ i : X,
163 finiteFoxStageDerivative (X := X) N n i u *
164 (MonoidAlgebra.of (ModNCompletedCoeff n)
165 (finiteFoxStageTargetQuotient (X := X) N)
166 (QuotientGroup.mk' N (FreeGroup.of i)) - 1)) +
167 gu * (∑ i : X,
168 finiteFoxStageDerivative (X := X) N n i v *
169 (MonoidAlgebra.of (ModNCompletedCoeff n)
170 (finiteFoxStageTargetQuotient (X := X) N)
171 (QuotientGroup.mk' N (FreeGroup.of i)) - 1)) := by
172 rw [hu, hv]
173 _ =
174 ∑ i : X,
175 (finiteFoxStageDerivative (X := X) N n i u +
176 gu * finiteFoxStageDerivative (X := X) N n i v) *
177 (MonoidAlgebra.of (ModNCompletedCoeff n)
178 (finiteFoxStageTargetQuotient (X := X) N)
179 (QuotientGroup.mk' N (FreeGroup.of i)) - 1) := by
180 rw [Finset.mul_sum, ← Finset.sum_add_distrib]
181 apply Finset.sum_congr rfl
182 intro i hi
183 noncomm_ring
184 _ =
185 ∑ i : X,

188 (finiteFoxStageTargetQuotient (X := X) N)
189 (QuotientGroup.mk' N (FreeGroup.of i)) - 1) := by

191 intro i hi

203/-- Conditional finite-stage Fox boundary formula. Any finite-stage crossed differential on the

205theorem finiteFoxStageFoxBoundary_of_crossedDifferential
206 [Fintype X]

208 (hdelta :

213 (w : FreeGroup X) :
214 finiteFoxStageFoxBoundary (X := X) N n (delta w) =
