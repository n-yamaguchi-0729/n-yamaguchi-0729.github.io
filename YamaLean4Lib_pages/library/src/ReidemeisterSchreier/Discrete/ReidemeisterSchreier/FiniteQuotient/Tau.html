27/-- Schreier rewriting of a raw word list, indexed by quotient states. -/
28def tauList : Q → List (X × Bool) → FreeGroup (FiniteSchreierSymbol X Q)
29 | _q, [] => 1
30 | q, (x, true) :: xs =>
31 FreeGroup.of (q, x) * tauList (D.transition q x) xs
32 | q, (x, false) :: xs =>
33 (FreeGroup.of (D.inverseTransition q x, x))⁻¹ *
34 tauList (D.inverseTransition q x) xs

37theorem tauList_cons_true_false
38 (q : Q) (x : X) (xs : List (X × Bool)) :
39 D.tauList q ((x, true) :: (x, false) :: xs) =
40 D.tauList q xs := by
41 simp only [tauList, transition_eq, inverseTransition_eq, mul_inv_cancel_right, mul_inv_cancel_left]

52 (q : Q) (x : X) (b : Bool) (xs : List (X × Bool)) :
53 D.tauList q ((x, b) :: (x, !b) :: xs) =

59theorem tauList_append
60 (q : Q) (xs ys : List (X × Bool)) :
61 D.tauList q (xs ++ ys) =
62 D.tauList q xs *
63 D.tauList
64 (D.quotientMap (D.quotientSection q * FreeGroup.mk xs)) ys := by
65 induction xs generalizing q with
66 | nil =>
67 rw [← FreeGroup.one_eq_mk]
68 simp only [List.nil_append, tauList, mul_one, quotientMap_quotientSection_apply, one_mul]
69 | cons hd xs ih =>
70 rcases hd with ⟨x, eps⟩
71 cases eps
72 · simp only [List.cons_append, tauList, inverseTransition_eq, ih, map_mul, quotientMap_quotientSection_apply,
73 mul_assoc, SchreierRewriting.mk_cons_false, map_inv]
74 · simp only [List.cons_append, tauList, transition_eq, ih, map_mul, quotientMap_quotientSection_apply,
75 mul_assoc, SchreierRewriting.mk_cons_true]

79 (h : FreeGroup.Red.Step xs ys) :
80 D.tauList q xs = D.tauList q ys := by
81 cases h with
82 | not =>

86theorem tauList_red
87 (q : Q) {xs ys : List (X × Bool)}
88 (h : FreeGroup.Red xs ys) :
89 D.tauList q xs = D.tauList q ys := by
90 induction h with
91 | refl =>
92 rfl
93 | tail _ hstep ih =>
94 exact ih.trans (D.tauList_redStep q hstep)

125/-- Finite-quotient Schreier rewriting of a free-group word. -/
126def tau (q : Q) (w : FreeGroup X) : FreeGroup (FiniteSchreierSymbol X Q) :=
127 D.tauList q w.toWord

129@[simp]
130theorem tau_one (q : Q) :
131 D.tau q 1 = 1 := by
132 simp only [tau, FreeGroup.toWord_one, tauList]

135theorem tau_of (q : Q) (x : X) :
136 D.tau q (FreeGroup.of x) = FreeGroup.of (q, x) := by

139theorem tau_mk (q : Q) (xs : List (X × Bool)) :
140 D.tau q (FreeGroup.mk xs) = D.tauList q xs := by

144theorem symbolEvalHom_tau (q : Q) (w : FreeGroup X) :
145 D.symbolEvalHom (D.tau q w) =
146 D.quotientSection q * w *
147 (D.quotientSection (D.quotientMap (D.quotientSection q * w)))⁻¹ := by
148 simp only [tau, symbolEvalHom_tauList, FreeGroup.mk_toWord, map_mul, quotientMap_quotientSection_apply]

150theorem tau_mul (q : Q) (u v : FreeGroup X) :
151 D.tau q (u * v) =
152 D.tau q u *
153 D.tau (D.quotientMap (D.quotientSection q * u)) v := by
154 calc
155 D.tau q (u * v) =
156 D.tau q (FreeGroup.mk (u.toWord ++ v.toWord)) := by
157 rw [← FreeGroup.mul_mk, FreeGroup.mk_toWord, FreeGroup.mk_toWord]
158 _ = D.tauList q (u.toWord ++ v.toWord) := D.tau_mk q (u.toWord ++ v.toWord)
159 _ =
160 D.tauList q u.toWord *
161 D.tauList (D.quotientMap (D.quotientSection q * FreeGroup.mk u.toWord))
162 v.toWord := by
163 rw [tauList_append]
164 _ =
165 D.tau q u *
166 D.tau (D.quotientMap (D.quotientSection q * u)) v := by
167 simp only [FreeGroup.mk_toWord, map_mul, quotientMap_quotientSection_apply, tau]

169theorem tau_mul_of_mem_kernel
170 {u v : FreeGroup X} (hu : u ∈ D.kernel) :
171 D.tau 1 (u * v) = D.tau 1 u * D.tau 1 v := by
172 have huquot : D.quotientMap u = 1 := by
173 simpa [kernel] using hu
174 simpa [D.quotientSection_one, huquot] using D.tau_mul 1 u v

176theorem tau_inv (q : Q) (u : FreeGroup X) :
177 D.tau (D.quotientMap (D.quotientSection q * u)) u⁻¹ =
178 (D.tau q u)⁻¹ := by
179 have hmul :
180 D.tau q u *
181 D.tau (D.quotientMap (D.quotientSection q * u)) u⁻¹ = 1 := by
182 simpa using (D.tau_mul q u u⁻¹).symm
183 calc
184 D.tau (D.quotientMap (D.quotientSection q * u)) u⁻¹ =
185 1 * D.tau (D.quotientMap (D.quotientSection q * u)) u⁻¹ := by
186 simp only [map_mul, quotientMap_quotientSection_apply, one_mul]
187 _ =
188 ((D.tau q u)⁻¹ * D.tau q u) *
189 D.tau (D.quotientMap (D.quotientSection q * u)) u⁻¹ := by
190 simp only [map_mul, quotientMap_quotientSection_apply, one_mul, inv_mul_cancel]
191 _ =
192 (D.tau q u)⁻¹ *
193 (D.tau q u *
194 D.tau (D.quotientMap (D.quotientSection q * u)) u⁻¹) := by
195 group
196 _ = (D.tau q u)⁻¹ := by
197 rw [hmul]
198 simp only [mul_one]

200theorem tau_inv_of_mem_kernel
201 {u : FreeGroup X} (hu : u ∈ D.kernel) :
202 D.tau 1 u⁻¹ = (D.tau 1 u)⁻¹ := by
203 have huquot : D.quotientMap u = 1 := by
204 simpa [kernel] using hu
205 simpa [D.quotientSection_one, huquot] using D.tau_inv 1 u
