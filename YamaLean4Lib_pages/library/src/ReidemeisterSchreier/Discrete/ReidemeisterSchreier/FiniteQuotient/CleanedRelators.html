28/-- The relator family obtained by rewriting original relators and then
29deleting degenerate Schreier generators. -/
30def cleanedSchreierRelators
31 (R : Set (FreeGroup X))
32 [DecidablePred D.IsDegenerateSchreierSymbol] :
33 Set (FreeGroup D.NondegenerateSchreierSymbol) :=
34 { z | ∃ q : Q, ∃ r ∈ R, z = D.cleanedTau q r }
35
36theorem mem_cleanedSchreierRelators
37 (R : Set (FreeGroup X))
38 [DecidablePred D.IsDegenerateSchreierSymbol]
39 (q : Q) {r : FreeGroup X} (hr : r ∈ R) :
40 D.cleanedTau q r ∈ D.cleanedSchreierRelators R :=
41 ⟨q, r, hr, rfl⟩

43noncomputable def cleanedSchreierRelatorFinset
44 [DecidablePred D.IsDegenerateSchreierSymbol]
45 [DecidableEq (FreeGroup D.NondegenerateSchreierSymbol)]
46 (R : Finset (FreeGroup X)) :
47 Finset (FreeGroup D.NondegenerateSchreierSymbol) :=
48 (Finset.univ : Finset Q).biUnion
49 (fun q => R.image (fun r => D.cleanedTau q r))

96/-- Relators forcing the chosen quotient representatives to rewrite to the
97empty word. These are automatic for a prefix-closed Schreier transversal, but
98are useful for arbitrary finite quotient sections. -/
99def quotientSectionRelators :
100 Set (FreeGroup (FiniteSchreierSymbol X Q)) :=
101 { q | ∃ s : Q, q = D.tau 1 (D.quotientSection s) }

103def augmentedPresentationRelators (R : Set (FreeGroup X)) :
104 Set (FreeGroup (FiniteSchreierSymbol X Q)) :=
105 D.presentationRelators R ∪ D.quotientSectionRelators
