25/-- A normalized representative function for right cosets of `L` in a free
26group. The condition `g * (representative g)⁻¹ ∈ L` says that
27`representative g` represents the same right coset as `g`. -/
28structure RightSchreierRepresentative (L : Subgroup (FreeGroup X)) where
29 representative : FreeGroup X → FreeGroup X
30 sameRightCoset : ∀ g, g * (representative g)⁻¹ ∈ L
31 representative_eq_of_sameRightCoset :
32 ∀ {g h}, g * h⁻¹ ∈ L → representative g = representative h
33 representative_one : representative 1 = 1

148/-- Schreier rewriting of a raw word list, starting from prefix `p`. -/
149def tauList : FreeGroup X → List (X × Bool) → FreeGroup (SchreierSymbol X)
150 | _p, [] => 1
151 | p, (x, true) :: xs =>
152 FreeGroup.of (S.representative p, x) *
153 tauList (p * FreeGroup.of x) xs
154 | p, (x, false) :: xs =>
155 (FreeGroup.of (S.representative (p * (FreeGroup.of x)⁻¹), x))⁻¹ *
156 tauList (p * (FreeGroup.of x)⁻¹) xs

159theorem symbolEvalHom_tauList
160 (p : FreeGroup X) (xs : List (X × Bool)) :
161 symbolEvalHom S (tauList S p xs) =
162 S.representative p * FreeGroup.mk xs *
163 (S.representative (p * FreeGroup.mk xs))⁻¹ := by
164 induction xs generalizing p with
165 | nil =>
166 simp only [tauList, map_one]

876current prefix is obtained from the previous representative by appending the
877positive letter. -/

880 | _p, [] => True
