29/-- A finite subfamily of a directed family of sets has a single lower bound in the
30`Directed (· ⊇ ·)` order. -/
31theorem directed_finset_has_lower_bound {α : Type*} {ι : Type*} {U : ι → Set α}
32 (hdir : Directed (· ⊇ ·) U) :
33 ∀ s : Finset ι, s.Nonempty → ∃ k : ι, ∀ i ∈ s, U k ⊆ U i := by
34 classical
35 intro s
36 refine Finset.induction_on s ?_ ?_
37 · intro hs
38 rcases hs with ⟨i, hi⟩
39 simp only [Finset.notMem_empty] at hi
40 · intro a s ha ih hs
41 by_cases hs' : s.Nonempty
42 · rcases ih hs' with ⟨j, hj⟩
43 rcases hdir a j with ⟨k, hka, hkj⟩
44 refine ⟨k, ?_⟩
45 intro i hi
46 rw [Finset.mem_insert] at hi
47 rcases hi with rfl | hi
48 · exact hka
49 · exact hkj.trans (hj i hi)
50 · have hs_singleton : s = ∅ := Finset.not_nonempty_iff_eq_empty.mp hs'
51 refine ⟨a, ?_⟩
52 intro i hi
53 have : i = a := by
54 simpa [hs_singleton] using hi
55 subst this
56 exact Subset.rfl

58/-- For a closed subgroup of a profinite group, multiplication by the
59intersection of a filtered family of closed sets is the intersection of the products. -/
60theorem closedSubgroup_mul_iInter_eq_iInter_mul [CompactSpace G]
61 (H : ClosedSubgroup G) {ι : Type v} (U : ι → Set G)
62 (hclosed : ∀ i, IsClosed (U i))
63 (hdir : Directed (· ⊇ ·) U) :
64 ((H : Set G) * ⋂ i, U i) = ⋂ i, ((H : Set G) * U i) := by
65 classical
66 ext x
67 constructor
68 · intro hx
69 rcases hx with ⟨h, hhH, u, hu, rfl⟩
70 refine Set.mem_iInter.2 ?_
71 intro i
72 exact ⟨h, hhH, u, Set.mem_iInter.1 hu i, rfl⟩
73 · intro hx
74 by_cases hι : Nonempty ι
75 · let A : ι → Set H := fun i => {h : H | ((h : G)⁻¹ * x) ∈ U i}
76 have hAclosed : ∀ i, IsClosed (A i) := by
77 intro i
78 change IsClosed (((fun h : H => ((h : G)⁻¹ * x)) ⁻¹' U i))
79 exact (hclosed i).preimage (continuous_subtype_val.inv.mul continuous_const)
80 have hAfin : ∀ s : Finset ι, (⋂ i ∈ s, A i).Nonempty := by
81 intro s
82 by_cases hs : s.Nonempty
83 · rcases directed_finset_has_lower_bound hdir s hs with ⟨k, hk⟩
84 rcases Set.mem_iInter.1 hx k with ⟨h, hhH, u, huK, hmul⟩
85 refine ⟨⟨h, hhH⟩, ?_⟩
86 refine Set.mem_iInter₂.2 ?_
87 intro i hi
88 change h⁻¹ * x ∈ U i
89 rw [← hmul]
90 simpa [mul_assoc] using hk i hi huK
91 · refine ⟨1, ?_⟩
92 simp only [Finset.not_nonempty_iff_eq_empty.mp hs, Finset.notMem_empty, iInter_of_empty, iInter_univ,
93 mem_univ]
94 have hA : (⋂ i, A i).Nonempty := by
95 simpa using
96 (isCompact_univ : IsCompact (Set.univ : Set H)).inter_iInter_nonempty A hAclosed
97 (fun s => by
98 simpa [Set.inter_univ] using hAfin s)
99 rcases hA with ⟨h, hh⟩
100 have hhU : ∀ i, ((h : G)⁻¹ * x) ∈ U i := by
101 intro i
102 exact Set.mem_iInter.1 hh i
103 refine ⟨(h : G), h.2, ((h : G)⁻¹ * x), Set.mem_iInter.2 hhU, by simp only [mul_inv_cancel_left]⟩
104 · have : IsEmpty ι := not_nonempty_iff.mp hι
105 have hxuniv : x ∈ ((H : Set G) * (Set.univ : Set G)) := by
106 exact ⟨1, H.one_mem, x, Set.mem_univ x, one_mul x⟩
107 simpa [Set.iInter_of_empty, this] using hxuniv

109/-- Continuous surjections commute with the intersection of a filtered
110family of closed sets in a compact domain. -/
111theorem continuous_surjective_image_iInter_eq_iInter_image {X : Type u}
112 [TopologicalSpace X] [CompactSpace X] {R : Type v} [TopologicalSpace R] [T2Space R]
113 (φ : X → R) (hφ : Continuous φ) (hsurj : Function.Surjective φ)
114 {ι : Type w} (U : ι → Set X)
115 (hclosed : ∀ i, IsClosed (U i))
116 (hdir : Directed (· ⊇ ·) U) :
117 φ '' ⋂ i, U i = ⋂ i, φ '' U i := by
118 classical
119 ext y
120 constructor
121 · rintro ⟨x, hx, rfl⟩
122 refine Set.mem_iInter.2 ?_
123 intro i
124 exact ⟨x, Set.mem_iInter.1 hx i, rfl⟩
125 · intro hy
126 by_cases hι : Nonempty ι
127 · let A : ι → Set X := fun i => U i ∩ φ ⁻¹' ({y} : Set R)
128 have hAclosed : ∀ i, IsClosed (A i) := by
129 intro i
130 exact (hclosed i).inter ((isClosed_singleton.preimage hφ))
131 have hAfin : ∀ s : Finset ι, (⋂ i ∈ s, A i).Nonempty := by
132 intro s
133 by_cases hs : s.Nonempty
134 · rcases directed_finset_has_lower_bound hdir s hs with ⟨k, hk⟩
135 rcases Set.mem_iInter.1 hy k with ⟨x, hxU, hxy⟩
136 refine ⟨x, Set.mem_iInter₂.2 ?_⟩
137 intro i hi
138 refine ⟨hk i hi hxU, ?_⟩
139 simpa [Set.mem_singleton_iff] using hxy
140 · rcases hsurj y with ⟨x, rfl⟩
141 refine ⟨x, ?_⟩
142 simp only [Finset.not_nonempty_iff_eq_empty.mp hs, Finset.notMem_empty, iInter_of_empty, iInter_univ,
143 mem_univ]
144 have hA : (⋂ i, A i).Nonempty := by
145 simpa using
146 (isCompact_univ : IsCompact (Set.univ : Set X)).inter_iInter_nonempty A hAclosed
147 (fun s => by
148 simpa [Set.inter_univ] using hAfin s)
149 rcases hA with ⟨x, hx⟩
150 refine ⟨x, ?_, ?_⟩
151 · exact Set.mem_iInter.2 fun i => (Set.mem_iInter.1 hx i).1
152 · simpa [Set.mem_singleton_iff] using (Set.mem_iInter.1 hx (Classical.choice hι)).2
153 · have : IsEmpty ι := not_nonempty_iff.mp hι
154 simpa [Set.iInter_of_empty, this] using hsurj y

156/-- Every open subgroup containing a closed subgroup also contains a set
157of the form `H U` with `U` open and normal. -/
158theorem exists_openNormalSubgroup_mul_subset_openSubgroup [CompactSpace G]
159 (H : ClosedSubgroup G) (V : OpenSubgroup G)
160 (hHV : (H : Subgroup G) ≤ (V : Subgroup G)) :
161 ∃ U : OpenNormalSubgroup G, ((H : Set G) * (U : Set G)) ⊆ (V : Set G) := by
162 have hVfin : Subgroup.FiniteIndex (V : Subgroup G) := V.finiteIndex_of_finite_quotient
163 let U : OpenNormalSubgroup G :=
164 { toSubgroup := Subgroup.normalCore V
165 isOpen' := Subgroup.isOpen_of_isClosed_of_finiteIndex _ (V.normalCore_isClosed V.isClosed) }
166 refine ⟨U, ?_⟩
167 intro x hx
168 rcases hx with ⟨h, hhH, u, huU, rfl⟩
169 exact V.mul_mem (hHV hhH) (V.normalCore_le huU)

171/-- A closed normal subgroup of a profinite group is the
172intersection of all open normal subgroups containing it. -/
173theorem closedSubgroup_eq_sInf_openNormal [CompactSpace G] [TotallyDisconnectedSpace G]
174 (H : ClosedSubgroup G) [((H : Subgroup G).Normal)] :
175 (H : Subgroup G) =
176 sInf {N : Subgroup G | IsOpen (N : Set G) ∧ (H : Subgroup G) ≤ N ∧ N.Normal} := by
177 ext x
178 constructor
179 · intro hx
180 simp only [Subgroup.mem_sInf, Set.mem_setOf_eq]
181 intro N hN
182 exact hN.2.1 hx
183 · intro hx
184 have hxall :
185 ∀ N : Subgroup G, IsOpen (N : Set G) ∧ (H : Subgroup G) ≤ N ∧ N.Normal → x ∈ N := by
186 simpa only [Subgroup.mem_sInf, Set.mem_setOf_eq] using hx
187 have hxOpen :
188 x ∈ sInf {N : Subgroup G | IsOpen (N : Set G) ∧ (H : Subgroup G) ≤ N} := by
189 simp only [Subgroup.mem_sInf, Set.mem_setOf_eq]
190 intro N hN
191 have hNfin : Subgroup.FiniteIndex N := by
192 letI : Finite (G ⧸ N) := Subgroup.quotient_finite_of_isOpen N hN.1
193 exact Subgroup.finiteIndex_of_finite_quotient
194 have hcoreOpen : IsOpen (N.normalCore : Set G) := by
195 exact Subgroup.isOpen_of_isClosed_of_finiteIndex _ (N.normalCore_isClosed
196 (Subgroup.isClosed_of_isOpen N hN.1))
197 have hHcore : (H : Subgroup G) ≤ N.normalCore := (Subgroup.normal_le_normalCore).2 hN.2
198 have hxcore : x ∈ N.normalCore := by
199 exact hxall N.normalCore ⟨hcoreOpen, hHcore, inferInstance⟩
200 exact N.normalCore_le hxcore
201 exact (closedSubgroup_eq_sInf_open (G := G) H) ▸ hxOpen

203end

210/-- If the intersection of a family of closed subgroups lies inside an
211open subgroup, then already a finite subfamily has the same property.
212
213We formulate the family as plain subgroups together with explicit closedness hypotheses in order
214to keep the later API flexible.
215-/
216theorem finite_iInter_subgroup_subset_openSubgroup [CompactSpace G] {ι : Type v}
217 (H : ι → Subgroup G)
218 (hclosed : ∀ i, IsClosed (((H i : Subgroup G) : Set G)))
219 (U : OpenSubgroup G)
220 (hInter : (⋂ i, (((H i : Subgroup G) : Set G))) ⊆ (((U : Subgroup G) : Set G))) :
221 ∃ s : Finset ι, (⋂ i ∈ s, (((H i : Subgroup G) : Set G))) ⊆ (((U : Subgroup G) : Set G)) := by
222 let K : Set G := (((U : Subgroup G) : Set G))ᶜ
223 have hKclosed : IsClosed K := by
224 simpa [K] using (openSubgroup_isOpen (G := G) U).isClosed_compl
225 have hKcompact : IsCompact K := hKclosed.isCompact
226 have havoid : K ∩ ⋂ i, (((H i : Subgroup G) : Set G)) = ∅ := by
227 ext x
228 constructor
229 · intro hx
230 exact False.elim (hx.1 (hInter hx.2))
231 · intro hx
232 simp only [mem_empty_iff_false] at hx
233 rcases hKcompact.elim_finite_subfamily_closed
234 (fun i => (((H i : Subgroup G) : Set G))) hclosed havoid with ⟨s, hs⟩
235 refine ⟨s, ?_⟩
236 intro x hx
237 by_contra hxU
238 have hxK : x ∈ K := by
239 simpa [K] using hxU
240 have hmem : x ∈ K ∩ ⋂ i ∈ s, (((H i : Subgroup G) : Set G)) := by
241 exact ⟨hxK, hx⟩
242 have : x ∈ (∅ : Set G) := by
243 simp only [hs, mem_empty_iff_false] at hmem
244 simp only [mem_empty_iff_false] at this

246/-- If a family of open subgroups of a profinite group has trivial total
247intersection, then finite intersections of members of the family form a neighborhood basis of `1`.
248-/
249theorem finite_openSubgroup_intersections_form_nhds_basis [CompactSpace G]
250 [IsTopologicalGroup G] [TotallyDisconnectedSpace G] {ι : Type v} (U : ι → OpenSubgroup G)
251 (hInter : (⋂ i, (((U i : Subgroup G) : Set G))) = ({1} : Set G)) :
252 ∀ W : Set G, IsOpen W → (1 : G) ∈ W →
253 ∃ s : Finset ι, (⋂ i ∈ s, (((U i : Subgroup G) : Set G))) ⊆ W := by
254 intro W hW h1W
255 rcases exists_openNormalSubgroup_sub_open_nhds_of_one (G := G) hW h1W with ⟨N, hNW⟩
256 have hInterN : (⋂ i, (((U i : Subgroup G) : Set G))) ⊆ (((N : Subgroup G) : Set G)) := by
257 intro x hx
258 have hx1 : x = 1 := by
259 have hxsingleton : x ∈ ({1} : Set G) := by
260 rw [← hInter]
261 exact hx
262 simpa using hxsingleton
263 simp only [OpenSubgroup.coe_toSubgroup, hx1, SetLike.mem_coe, one_mem]
264 rcases finite_iInter_subgroup_subset_openSubgroup (G := G)
265 (fun i => (U i : Subgroup G))
266 (fun i => openSubgroup_isClosed (G := G) (U i))
267 N.toOpenSubgroup hInterN with ⟨s, hs⟩
268 exact ⟨s, hs.trans hNW⟩
