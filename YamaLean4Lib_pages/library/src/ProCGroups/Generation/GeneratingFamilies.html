23/-- An indexed topological generating family converging to `1`.
24
25This is deliberately weaker than a free pro-`C` basis: it records only generation and convergence
26of the image. -/
27structure TopologicalGeneratingFamily
28 (G : Type u) [Group G] [TopologicalSpace G] [IsTopologicalGroup G] where
29 index : Type u
30 toFun : index → G
31 convergesToOne : _root_.ProCGroups.FreeProC.FamilyConvergesToOne (G := G) toFun
32 generates : TopologicallyGenerates (G := G) (Set.range toFun)
33
34namespace TopologicalGeneratingFamily
35
36variable {G : Type u} [Group G] [TopologicalSpace G] [IsTopologicalGroup G]
37
38instance instCoeFunTopologicalGeneratingFamily :
39 CoeFun (TopologicalGeneratingFamily G) (fun family => family.index → G) where
40 coe family := family.toFun
41
42@[simp] theorem toFun_eq_coe (family : TopologicalGeneratingFamily G) :
43 family.toFun = (family : family.index → G) := rfl
44
45/-- The cardinality of the indexing type. -/
46def cardinal (family : TopologicalGeneratingFamily G) : Cardinal :=
47 Cardinal.mk family.index

49end TopologicalGeneratingFamily
