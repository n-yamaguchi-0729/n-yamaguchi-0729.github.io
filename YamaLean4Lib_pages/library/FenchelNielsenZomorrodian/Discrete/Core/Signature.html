Definition FenchelNielsen.FenchelSignature.eulerCharacteristic Source 
def FenchelSignature.eulerCharacteristic (σ : FenchelSignature) : ℚ :=
  (2 : ℚ) - 2 * σ.orbitGenus - σ.numCusps -
    ∑ i : Fin σ.numPeriods, (1 - ((σ.periods i : ℚ)⁻¹))
