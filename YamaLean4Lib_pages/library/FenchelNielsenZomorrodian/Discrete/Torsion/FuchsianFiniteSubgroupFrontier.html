def subgroup_le_conj_ellipticStabilizer
    (σ : FuchsianSignature) (K : Subgroup (FuchsianPresentedGroup σ)) : Prop :=
  ∃ i : Fin σ.numPeriods, ∃ c : FuchsianPresentedGroup σ,
    ∀ k : K, ∃ n : ℤ,
      (k : FuchsianPresentedGroup σ) =
        c * ellipticElement σ i ^ n * c⁻¹
