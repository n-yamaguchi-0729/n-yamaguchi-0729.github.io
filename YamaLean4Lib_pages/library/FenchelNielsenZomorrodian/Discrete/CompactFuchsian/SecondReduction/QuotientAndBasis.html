  toFun
    | ⟨.inl h, k⟩ => .inr (k, .inr (.inr h))
    | ⟨.inr (.inl d), _⟩ => .inl d
    | ⟨.inr (.inr (.inl r)), k⟩ => .inr (k, .inl r)
    | ⟨.inr (.inr (.inr jk)), k⟩ => .inr (k, .inr (.inl (jk.2, jk.1)))

    | .inl d => ⟨.inr (.inl d), (0 : Fin 1)⟩

    rcases x with ⟨src, k⟩
    cases src with
    | inl _ => rfl

        | inl _ =>

            rfl
        | inr rest =>

            | inl _ => rfl
            | inr _ => rfl

    | inl _ => rfl
