def firstReductionTailSuccEquivFirstSecondTail
    (tailLen p : ℕ) :
    Fin (tailLen + 1) × Fin p ≃ Sum (Fin p) (Fin tailLen × Fin p) where
  toFun jk :=
    Fin.cases (motive := fun _ => Sum (Fin p) (Fin tailLen × Fin p))
      (.inl jk.2)
      (fun j => .inr (j, jk.2))
      jk.1
  invFun
    | .inl k => (0, k)
    | .inr jk => (jk.1.succ, jk.2)
  left_inv := by
    intro jk
    rcases jk with ⟨j, k⟩
    cases j using Fin.cases with
    | zero => rfl
    | succ j => rfl
  right_inv := by
    intro s
    cases s with
    | inl k => rfl
    | inr jk =>
        rcases jk with ⟨j, k⟩
        rfl
