def AbelianizationKernelConditionFamily {ι : Type*} [Fintype ι] [DecidableEq ι]
    (periods : ι → ℕ) : Prop :=
  ∀ i, periods i ∣ otherPeriodsLcmFamily periods i ∨
    2 * otherPeriodsLcmFamily periods i ≤ otherPeriodsProductFamily periods i
