Definition ReidemeisterSchreier.SchreierOrientation.cocycle GitHub 
def cocycle {Q : Type u} (o : SchreierOrientation)
    (sec : Q → F) (next : Q → X → Q) (ι : X → F) (q : Q) (x : X) : F :=
  match o with
  | left => (sec (next q x))⁻¹ * sec q * ι x
  | right => sec q * ι x * (sec (next q x))⁻¹
