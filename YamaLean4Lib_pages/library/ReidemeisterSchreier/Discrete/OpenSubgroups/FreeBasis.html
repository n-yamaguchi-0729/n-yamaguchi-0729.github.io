      Set (Quiver.Total

      intro a b e he
      exact schreierLabelFunctor_map_of_eq_one_of_mem_tree (X := X) hT e he

      (T := schreierPrefixTree (X := X) hT)

        intro a b e he
        exact htree e he)

      (IsFreeGroupoid.of i.1.hom)

      change loop.1 = (1 : FreeGroup X) * loop.1 * (1 : FreeGroup X)⁻¹
